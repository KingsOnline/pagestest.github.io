/* Makes bottom bar visible on load instead of after selecting central play button.
   This means students can see how long the video is before they begin playing it.
   All controls become responsive including the preview pane when you drag the scrubber so you can start the video from any point you like as well. */
.playkit-player.playkit-pre-playback .playkit-player-gui .playkit-gui-area .playkit-bottom-bar {
  opacity: 1;
  display: block;
  visibility: visible;
  max-height: 60px;
  transition: 100ms max-height ease-in-out, 100ms visibility ease-in-out;
  padding: 0 1rem 0.25rem;
}

/* Hides the central play button from the player when it loads.
   This means visual designers don't have to design thumbnails around the play button. */
button.playkit-pre-playback-play-button {
  display: none;
}

/* Makes the central play button visible when the video has ended.
   This makes it easier to replay the video, if needed, but also helps confirm to students that the video has ended. */
button.playkit-pre-playback-play-button[aria-label="Start over"] {
  display: block;
}

/* The following code is to ensure the play/pause button doesn't get hidden when the small or extra small class are active.
   This ensures that the play/pause button remains visible for mobile devices. */
.playkit-player.playkit-size-sm:not(.playkit-state-idle) .playkit-bottom-bar .playkit-playback-controls ,
.playkit-player.playkit-size-xs:not(.playkit-state-idle) .playkit-bottom-bar .playkit-playback-controls ,

  display: inline;
}
.playkit-player.playkit-size-sm:not(.playkit-state-idle) .playkit-bottom-bar .playkit-playback-controls .playkit-control-button-container,
.playkit-player.playkit-size-xs:not(.playkit-state-idle) .playkit-bottom-bar .playkit-playback-controls .playkit-control-button-container,
.playkit-player.playkit-touch:not(.playkit-state-idle) .playkit-bottom-bar .playkit-playback-controls .playkit-control-button-container {
  display: inline;
}
.playkit-player.playkit-size-sm:not(.playkit-state-idle) .playkit-bottom-bar .playkit-playback-controls .playkit-control-button-container .playkit-tooltip ,
.playkit-player.playkit-size-xs:not(.playkit-state-idle) .playkit-bottom-bar .playkit-playback-controls .playkit-control-button-container .playkit-tooltip ,
.playkit-player.playkit-touch:not(.playkit-state-idle) .playkit-bottom-bar .playkit-playback-controls .playkit-control-button-container .playkit-tooltip {
